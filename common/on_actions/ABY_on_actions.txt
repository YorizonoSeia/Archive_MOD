	on_startup = {
		effect = {
			ABY = {
				setup_central_map = yes
			}
		}
	}
	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = { 
						tag = ABY
					}
					FROM = { 
						original_tag = HYA
					}
				}
				set_global_flag = ABY_focus_operation_falling_flower_flag
				ROOT = {
					add_ideas = {
						ABY_idea_sword_to_bohemia
					}
					if = {
						limit = { has_country_flag = ABY_flag_czech_resistance }
						9 = {
							add_dynamic_modifier = {
								modifier = ABY_modifier_czech_resistance
								days = 30
								scope = HYA
							}
						}
					}
				}
				if = {
					limit = {
						OR = {
							has_country_flag = ABY_flag_german_resistance_1
							has_country_flag = ABY_flag_german_resistance_2
							has_country_flag = ABY_flag_german_resistance_3
						}
						FROM = {
							OR = {
								NOT = { has_global_flag = ABY_idea_HYA_suspecious_flag }
								has_idea = ABY_idea_HYA_suspecious
							}
							controls_state = 74
						}
					}
					FROM = { country_event = { id = ABY_new_event.201 hours = 8 } }
				}
			}
			if = {
				limit = {
					ROOT = { 
						original_tag = HYA
					}
					FROM = { 
						tag = ABY
					}
				}
				set_global_flag = ABY_focus_operation_falling_flower_flag
				FROM = {
					add_ideas = {
						ABY_idea_sword_to_bohemia
					}
					if = {
						limit = { has_country_flag = ABY_flag_czech_resistance }
						9 = {
							add_dynamic_modifier = {
								modifier = ABY_modifier_czech_resistance
								days = 30
								scope = HYA
							}
						}
					}
				}
				ROOT = {
					if = {
						limit = { 
							OR = {
								has_country_flag = ABY_flag_german_resistance_1
								has_country_flag = ABY_flag_german_resistance_2
								has_country_flag = ABY_flag_german_resistance_3
							}
							OR = {
								NOT = { has_global_flag = ABY_idea_HYA_suspecious_flag }
								has_idea = ABY_idea_HYA_suspecious
							}
							controls_state = 74
						}
						country_event = { id = ABY_new_event.201 hours = 8 }
					}
				}
			}
		}
	}
